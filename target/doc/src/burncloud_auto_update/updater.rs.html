3use crate::{UpdateConfig, UpdateError, UpdateResult};
4use log::{info, error};
5use self_update::backends::github;

7/// 自动更新器
8pub struct AutoUpdater {
9    config: UpdateConfig,
10}
11
12impl AutoUpdater {
13    /// 创建新的自动更新器
14    pub fn new(config: UpdateConfig) -> Self {
15        Self { config }
16    }
17
18    /// 使用默认配置创建自动更新器
19    pub fn with_default_config() -> Self {
20        Self::new(UpdateConfig::default())
21    }
22
23    /// 检查是否有可用更新
24    pub async fn check_for_updates(&self) -> UpdateResult<bool> {
25        info!("检查更新中...");
26
27        match self.check_github_updates().await {
28            Ok(has_update) => Ok(has_update),
29            Err(e) => {
30                error!("GitHub 检查更新失败: {}", e);
31                Err(e)
32            }
33        }
34    }
35
36    /// 执行更新
37    pub async fn update_with_fallback(&self) -> UpdateResult<()> {
38        info!("开始更新应用程序...");
39
40        match self.update_from_github().await {
41            Ok(_) => {
42                info!("从 GitHub 更新成功");
43                Ok(())
44            }
45            Err(e) => {
46                error!("GitHub 更新失败: {}", e);
47                Err(e)
48            }
49        }
50    }
51
52    /// 从 GitHub 检查更新
53    async fn check_github_updates(&self) -> UpdateResult<bool> {
54        info!("正在检查 GitHub 更新...");
55
56        let update = github::Update::configure()?
57            .repo_owner(&self.config.github_owner)
58            .repo_name(&self.config.github_repo)
59            .bin_name(&self.config.bin_name)
60            .current_version(&self.config.current_version)
61            .build()
62            .map_err(UpdateError::from)?;
63
64        let latest_release = github::Update::get_latest_release(&update)
65            .map_err(UpdateError::from)?;
66        let current_version = &self.config.current_version;
67
68        info!("当前版本: {}, 最新版本: {}", current_version, latest_release.version);
69
70        Ok(latest_release.version != *current_version)
71    }
72
73    /// 从 GitHub 更新
74    async fn update_from_github(&self) -> UpdateResult<()> {
75        info!("正在从 GitHub 下载更新...");
76
77        let update = github::Update::configure()?
78            .repo_owner(&self.config.github_owner)
79            .repo_name(&self.config.github_repo)
80            .bin_name(&self.config.bin_name)
81            .current_version(&self.config.current_version)
82            .build()
83            .map_err(UpdateError::from)?;
84
85        let status = update.update()
86            .map_err(UpdateError::from)?;
87
88        match status.updated() {
89            true => {
90                info!("更新成功，新版本: {}", status.version());
91                Ok(())
92            }
93            false => {
94                info!("已是最新版本");
95                Ok(())
96            }
97        }
98    }
99
100    /// 获取当前版本
101    pub fn current_version(&self) -> &str {
102        &self.config.current_version
103    }
104
105    /// 设置新的配置
106    pub fn set_config(&mut self, config: UpdateConfig) {
107        self.config = config;
108    }
109
110    /// 获取配置的引用
111    pub fn config(&self) -> &UpdateConfig {
112        &self.config
113    }
114
115    /// 获取手动下载链接
116    pub fn get_download_links(&self) -> (String, String) {
117        self.config.download_links()
118    }
119

177#[cfg(test)]
178mod tests {
179    use super::*;

181    #[test]
182    fn test_auto_updater_creation() {
183        let updater = AutoUpdater::with_default_config();
184        assert_eq!(updater.current_version(), std::env!("CARGO_PKG_VERSION"));
185    }
186
187    #[test]
188    fn test_get_download_links() {
189        let updater = AutoUpdater::with_default_config();
190        let (github_url, gitee_url) = updater.get_download_links();
191
192        assert_eq!(github_url, "https://github.com/burncloud/burncloud/releases");
193        assert_eq!(gitee_url, "https://gitee.com/burncloud/burncloud/releases");
194    }
195
196    #[test]
197    fn test_update_config_customization() {
198        let mut config = UpdateConfig::default();
199        config.current_version = "1.0.0".to_string();
200
201        let mut updater = AutoUpdater::new(config);
202        assert_eq!(updater.current_version(), "1.0.0");
203
204        let new_config = UpdateConfig {
205            current_version: "2.0.0".to_string(),
206            ..UpdateConfig::default()
207        };
208
209        updater.set_config(new_config);
210        assert_eq!(updater.current_version(), "2.0.0");
211    }
212
213    #[test]
214    fn test_clone() {
215        let updater = AutoUpdater::with_default_config();
216        let cloned = updater.clone();
217
218        assert_eq!(updater.current_version(), cloned.current_version());
219        assert_eq!(updater.config().github_owner, cloned.config().github_owner);
220    }

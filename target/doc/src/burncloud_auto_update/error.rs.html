1//! 自动更新错误类型
2
3use std::fmt;
4
5/// 自动更新错误类型
6#[derive(Debug)]
7pub enum UpdateError {
8    /// 网络错误
9    Network(String),
10    /// GitHub API 错误
11    GitHub(String),
12    /// 版本解析错误
13    Version(String),
14    /// 文件系统错误
15    FileSystem(String),
16    /// 权限错误
17    Permission(String),
18    /// 配置错误
19    Configuration(String),
20    /// 未知错误
21    Unknown(String),
22}
23
24impl fmt::Display for UpdateError {
25    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
26        match self {
27            UpdateError::Network(msg) => write!(f, "网络错误: {}", msg),
28            UpdateError::GitHub(msg) => write!(f, "GitHub API 错误: {}", msg),
29            UpdateError::Version(msg) => write!(f, "版本解析错误: {}", msg),
30            UpdateError::FileSystem(msg) => write!(f, "文件系统错误: {}", msg),
31            UpdateError::Permission(msg) => write!(f, "权限错误: {}", msg),
32            UpdateError::Configuration(msg) => write!(f, "配置错误: {}", msg),
33            UpdateError::Unknown(msg) => write!(f, "未知错误: {}", msg),
34        }
35    }
36}
37
38impl std::error::Error for UpdateError {}
39
40impl From<anyhow::Error> for UpdateError {
41    fn from(error: anyhow::Error) -> Self {
42        UpdateError::Unknown(error.to_string())
43    }
44}
45
46impl From<self_update::errors::Error> for UpdateError {
47    fn from(error: self_update::errors::Error) -> Self {
48        match error {
49            self_update::errors::Error::Network(_) => {
50                UpdateError::Network(error.to_string())
51            }
52            self_update::errors::Error::Release(_) => {
53                UpdateError::GitHub(error.to_string())
54            }
55            _ => UpdateError::Unknown(error.to_string()),
56        }
57    }
58}
59
60/// 自动更新结果类型
61pub type UpdateResult<T> = Result<T, UpdateError>;
